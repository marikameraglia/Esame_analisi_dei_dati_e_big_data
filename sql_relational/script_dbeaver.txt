###CREAZIONE TABELLE###
#Tabella CATEGORIA
CREATE TABLE CATEGORIA (
    IdCategoria INT PRIMARY KEY,
    NomeCategoria VARCHAR(100) NOT NULL); 

#Tabella SOTTOCATEGORIE
CREATE TABLE SOTTOCATEGORIA (
    IdSottocategoria INT PRIMARY KEY,
    NomeSottocategoria VARCHAR(100) NOT NULL,
    IdCategoria INT NOT NULL, -- Questa colonna conterrà l'ID della macro-categoria
    
    -- Definiamo il vincolo di Chiave Esterna
    CONSTRAINT fk_macro_categoria 
    FOREIGN KEY (IdCategoria) REFERENCES CATEGORIA(IdCategoria)
    ON DELETE CASCADE ON UPDATE CASCADE
);

#Tabella PRODOTTO
CREATE TABLE PRODOTTO (
    CodiceProd VARCHAR(50) PRIMARY KEY,
    Nome VARCHAR(150) NOT NULL,
    QuantitaDisponibile INT NOT NULL CHECK (QuantitaDisponibile >= 0),
    PrezzoUnitario DECIMAL(10,2) NOT NULL,
    Peso DECIMAL(6,2) NOT NULL,
    IdSottocategoria INT,
    FOREIGN KEY (IdSottocategoria) REFERENCES SOTTOCATEGORIA (IdSottocategoria) ON DELETE SET NULL
);


#Tabella CLIENTE
CREATE TABLE CLIENTE (
    IdCliente INT PRIMARY KEY,
    Nome VARCHAR(100) NOT NULL,
    Cognome VARCHAR(100) NOT NULL,
    Email VARCHAR(150) UNIQUE,
    Via VARCHAR(150),   -- Componente Indirizzo
    CAP VARCHAR(10),     -- Componente Indirizzo
    Citta VARCHAR(100)   -- Componente Indirizzo
);

#Tabella CORRIERE
CREATE TABLE CORRIERE (
    IdCorriere INT PRIMARY KEY,
    Nome VARCHAR(100) NOT NULL,
    Telefono VARCHAR(20)
);

#Tabella AREA COPERTURA
-- (Regola di mappatura: PK composta da FK del corriere + il valore dell'area)
CREATE TABLE AREA_COPERTURA_CORRIERE (
    IdCorriere INT,
    AreaCopertura VARCHAR(150),
    PRIMARY KEY (IdCorriere, AreaCopertura),
    FOREIGN KEY (IdCorriere) REFERENCES CORRIERE(IdCorriere) ON DELETE CASCADE
);

#Tabella SPEDIZIONE
CREATE TABLE SPEDIZIONE (
    IdSpedizione INT PRIMARY KEY,
    DataPartenza DATE,
    DataConsegnaPrevista DATE,
    StatoSpedizione VARCHAR(50),
    IdCorriere INT,
    FOREIGN KEY (IdCorriere) REFERENCES CORRIERE(IdCorriere) ON DELETE RESTRICT,
    CONSTRAINT chk_date_spedizione CHECK (DataConsegnaPrevista >= DataPartenza)
);

#Tabella ORDINE 
CREATE TABLE ORDINE (
    IdOrdine INT PRIMARY KEY,
    DataOrdine DATE NOT NULL,
    StatoOrdine VARCHAR(50),
    IdCliente INT NOT NULL, -- Un ordine deve appartenere obbligatoriamente a un cliente
    IdSpedizione INT,       -- Questo invece REA giusto lasciarlo NULL, perché la spedizione avviene dopo
    FOREIGN KEY (IdCliente) REFERENCES CLIENTE(IdCliente) ON DELETE CASCADE,
    FOREIGN KEY (IdSpedizione) REFERENCES SPEDIZIONE(IdSpedizione) ON DELETE SET NULL
);

#Tabella RIGA_ORDINE
CREATE TABLE RIGA_ORDINE (
    IdOrdine INT,
    CodiceProd VARCHAR(50),
    Quantita INT NOT NULL CHECK (Quantita > 0),
    PrezzoApplicato DECIMAL(10,2) NOT NULL,
    PRIMARY KEY (IdOrdine, CodiceProd),
    FOREIGN KEY (IdOrdine) REFERENCES ORDINE(IdOrdine) ON DELETE CASCADE,
    FOREIGN KEY (CodiceProd) REFERENCES PRODOTTO(CodiceProd) ON DELETE RESTRICT
);
